function rgb(r, g, b){
  let arr = [r, g, b];
  let resColorCode = '';
  
  arr = arr.map(function(value) {
    if(value > 255) return 255;
    else if(value < 0) return 0;
    else return value;
  });
  
  
  arr.forEach(value => {
    let hexVal = value.toString(16);
    if(hexVal.length == 1) resColorCode += '0';
    
    resColorCode += hexVal.toUpperCase();
  });
  return resColorCode;
}
